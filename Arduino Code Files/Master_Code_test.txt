#include <SoftwareSerial.h>


// inputs and outputs 
const int STARTBAND = 2;                    // Start the band of slave skulls                   
const int RESETSFX = 3;                     // Reset soundboard
const int LIGHTSOFF = 5;                    // Reset lights slave arduino to off
const int PLAY1 = 6;                        // Start soundboard
const int STOPSWITCH = 7;                   // Stopswitch
const int STARTSWITCH = 8;                  // Startswitch
const int INDICATOR = 12;                   // Optional Indicator for testing
unsigned long runtime;          
unsigned long offsetMillis;


void setup () {
// set the modes for the switches / pins
  pinMode(STARTSWITCH, INPUT_PULLUP);
  pinMode(STOPSWITCH, INPUT_PULLUP);
  pinMode(PLAY1, OUTPUT);
  pinMode(RESETSFX, OUTPUT);
  pinMode(STARTBAND, OUTPUT);
  pinMode(LIGHTSOFF, OUTPUT);
  pinMode(INDICATOR, OUTPUT);
  

// Preset Pin Signals
  digitalWrite(STARTBAND, HIGH);      // Activates slave skulls when LOW
  digitalWrite(RESETSFX, HIGH);       // Resets sound board when LOW
  digitalWrite(PLAY1, HIGH);          // Activates sound board when LOW
  digitalWrite(LIGHTSOFF, LOW);       // Triggers lights off on lights slave when HIGH (through transistor)
  digitalWrite(INDICATOR, HIGH);      // Optional indicator for testing     


// Serial COM setup
  Serial.begin(9600);
  Serial.setTimeout(10);

  
// Start Switch Option (for testing and programming) - will not be used in continuous play mode    
  while (digitalRead(STARTSWITCH) == HIGH) {}
  offsetMillis = millis();

}   // void setup
