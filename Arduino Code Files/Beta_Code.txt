// NeoPixel
#define PIN  2
#define NUMPIXELS 2
Adafruit_NeoPixel eyes = Adafruit_NeoPixel(NUMPIXELS, PIN, NEO_GRB + NEO_KHZ800);

void setup() 
{

EI=50;  // Brightness =50
  // Pin Mode Selection
pinMode(startpin, INPUT_PULLUP);
pinMode (resetPin, INPUT_PULLUP);

// LED Initialising
eyes.begin(); 
// Preset eyes to off
eyes.setPixelColor(0, eyes.Color(0,0,0)); // Eyes off
eyes.setPixelColor(1, eyes.Color(0,0,0)); // Eyes off
eyes.show(); // This sends the updated pixel color to the hardware.

servo1.attach(servoPin1);
servo1.write(120,10,false); // NOD  MAX UP 70   --MID 100--  MAX DOWN 130   
servo2.attach(servoPin2);
servo2.write(100,40,false);  // SHAKE  MAX STBD 30   --MID 90-- MAX PORT 170
servo3.attach(servoPin3);
servo3.write(95,40,false); // TILT  MAX STBD 140   --MID 90--  MAX PORT 50
servo4.attach(servoPin4);
servo4.write(47,40,false); // JAW  MAX OPEN 95   --MID 70--  MAX CLOSED 47  
servo5.attach(servoPin5);
servo5.write(80,40,false); // Eye X  MAX STBD 40 & Y70   --MID 80 --  MAX PORT 130  & Y70
servo6.attach(servoPin6);
servo6.write(80,40,false); /// Eye Y  MAX UP 120   --MID 80 --  MAX Down 30

delay(200);
// Start Position
servo1.write(130,15,false);
servo6.write(30,30,false);
servo2.write(80,20,false);

delay(1400);
delay(400);
// Wait for start signal
while (digitalRead(startpin) == HIGH) {};

// Define Offset millis
offsetMillis = millis();
}
