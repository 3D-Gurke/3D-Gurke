// NeoPixel
#define PIN  2
#define NUMPIXELS 2
Adafruit_NeoPixel eyes = Adafruit_NeoPixel(NUMPIXELS, PIN, NEO_GRB + NEO_KHZ800);

void setup() 
{
EI=128; // Set Eye brightness 0-256
  // Pin Mode Selection
pinMode(startpin, INPUT);
pinMode (resetPin, INPUT_PULLUP);

// LED Initialising
eyes.begin(); 
// Preset eyes to off
eyes.setPixelColor(0, eyes.Color(0,0,0)); // Eyes off
eyes.setPixelColor(1, eyes.Color(0,0,0)); // Eyes off
eyes.show(); // This sends the updated pixel color to the hardware.

// all servos straight
servo1.attach(servoPin1);
servo1.write(40,10,false); // // NOD  MAX UP 20   --MID 40--  MAX DOWN 80  
servo2.attach(servoPin2);
servo2.write(100,40,false); //SHAKE  MAX STBD 40   --MID 100--  MAX PORT 160
servo3.attach(servoPin3);
servo3.write(55,40,false); // TILT  MAX STBD 110   --MID 50--  MAX PORT 5
servo4.attach(servoPin4);
servo4.write(17,40,false); // JAW  MAX OPEN 60   --MID 30--    MAX CLOSED 13 
servo5.attach(servoPin5);
servo5.write(80,40,false); // Eye X  MAX STBD 30 & 85 Y   --MID 80--  MAX PORT 140  & 85 Y
servo6.attach(servoPin6);
servo6.write(90,40,false); //  Eye Y  MAX UP 130   --MID 90--  MAX Down 50
delay(200);
// Start Position
servo1.write(80,10,false); // // NOD  MAX UP 20   --MID 40--  MAX DOWN 80
servo6.write(50,40,false); //  Eye Y  MAX UP 130   --MID 90--  MAX Down 50 
delay(1400);
delay(400);

// Wait for start signal
while (digitalRead(startpin) == HIGH) {};



// Define Offset millis
offsetMillis = millis();
}
